ГСЧ = Новый ГенераторСлучайныхЧисел();
КоличествоЭлементовМассива = 10;
МассивА = Новый Массив(КоличествоЭлементовМассива);

ТекущееМаксЗначение = 0;
ТекущееМинЗначение = 10;

ИндексМинЗначения = 0;
ИндексМаксЗначения = 0;

СуммаЭлементов = 0;

Для Шаг = 0 По (КоличествоЭлементовМассива - 1) Цикл
	МассивА[Шаг] = ГСЧ.СлучайноеЧисло(0, 10);
	Сообщить("Массив " + Шаг + ": " + МассивА[Шаг]);
КонецЦикла;

Для Счетчик = 0  По (КоличествоЭлементовМассива - 1) Цикл
	
	Если (МассивА[Счетчик] < ТекущееМинЗначение) Тогда
		ТекущееМинЗначение = МассивА[Счетчик];
		ИндексМинЗначения = Счетчик;
	КонецЕсли;
	
	Если (МассивА[Счетчик] > ТекущееМаксЗначение) Тогда
		ТекущееМаксЗначение = МассивА[Счетчик];
		ИндексМаксЗначения = Счетчик;
	КонецЕсли;
	
КонецЦикла;

Для Счетчик = (ИндексМинЗначения + 1) По (ИндексМаксЗначения - 1) Цикл
	СуммаЭлементов = СуммаЭлементов + МассивА[Счетчик];		
КонецЦикла;

Сообщить("Минимальный индекс: " + ИндексМинЗначения);
